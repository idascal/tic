  3. Se citeşte din fişierul “vector.txt” un vector de numere naturale, n se află pe primul rând(n<1000), iar cele n elemente ale vectorului se află pe al doilea rând. Afişaţi câte numere perfecte sunt în vector. Afişarea se face în fişierul “solutie.out”. 

    
#include <iostream>
#include <fstream>
using namespace std;

ifstream fin ("vector.txt");
ofstream gout("solutie.out");

int perfect(int nr)
{
    int s=0;
    for(int i=1;i<nr;i++)
        if(nr%i==0)
            s+=i;
    if(s==nr)
        return 1;
    return 0;
}

int main()
{
    int n, r=0;
    fin>>n;
    int v[n+1];
    for(int i = 1;i<=n;i++)
    {
        fin>>v[i];
        if(perfect(v[i]))
            r++;
    }
    gout<<r;
    fin.close();
    gout.close();
    return 0;
}
