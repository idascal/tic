Program cu animaţie:

            #include <graphics.h>
            #include <conio.h>

            int main(int argc,char* argv[])
            {
                initwindow(800,600);
                settextstyle(DEFAULT_FONT,HORIZ_DIR,2);
                outtextxy(25,240,(char*)"Apasa orice tasta pentru animatie:");
                int i=0,j=3;
                getevent();

                while( i <= 420)
                {
                    setcolor(j);
                    rectangle(50+i,275,150+i,400);
                    rectangle(150+i,350,200+i,400);
                    circle(75+i,410,10);
                    circle(175+i,410,10);

                    delay(50);
                    j++;
                    if ( j == 15 )
                        j = 2;
                    i = i + 10;

                    refresh();
                    cleardevice();// clear screen
                }

                getch();
                closegraph();
                return 0;
            }
